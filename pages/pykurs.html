Lister og løkker

Dette er mye på en gang, men vi setter av tid til å jobbe med emnet 
talliste = ['a','b','c','d'] # lister har alltid []
print("Lista har", len(talliste), "elementer.")
for a in range (len(talliste)): # len er lengden av lista
  print('Element nr',a,'er:',talliste[a])
print('--------------------------------')
# Dette er en kommentar, det er bare en huskelapp til meg selv
# Loopen under legger til 4 tall på lista
for a in range (len(talliste)):
  print('Dette er a:',a)
  talliste.append(a) # legger til bakerst i lista
print(talliste)
print(talliste[5]) # printer element nr 5

If, elif, else

Også kjent som betingelser 
avstand = 78
if avstand > 100:
  print("Du er for langt unna!")
else:
  print("Du er nærme nok.")

temperatur = 62
if temperatur > 100:
  print("Vannet er damp.")
elif temperatur > 0:
  print("Vannet er flytende.")
else:
  print("Vannet er is.")
